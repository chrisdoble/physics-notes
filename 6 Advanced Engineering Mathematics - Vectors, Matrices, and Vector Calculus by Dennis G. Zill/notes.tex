\subsection{Vectors in 2-Space}

\begin{itemize}

  \item The vectors $\textbf{i}$ and $\textbf{j}$ are known as the \textbf{standard basis vectors} for $\mathbb{R}^2$
\end{itemize}

\subsection{Vectors in 3-Space}

\begin{itemize}

\end{itemize}
